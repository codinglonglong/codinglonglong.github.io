package xiangmu;
import java.util.Scanner;
public class Ceshi {
    public static void main(String args[]) {
        // 连续输入多个正整数，找到最大值
        int shuzi = 0;
        int zuidazhi = -1;
        Scanner sc = new Scanner(System.in);
        while(true) {
            System.out.print("请输入一个正整数(-1结束输入)：");
            shuzi = sc.nextInt();
            if(shuzi == -1) {
                break;
            }
            if(shuzi > zuidazhi) {
                zuidazhi = shuzi;
            }
        }
        System.out.println("最大值为：" + zuidazhi);
        sc.close();
    }
}
