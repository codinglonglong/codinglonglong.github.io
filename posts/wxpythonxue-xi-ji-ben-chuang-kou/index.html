    龙龙 2017-10-08 18:20

    import wx

class MyFrame1(wx.Frame):

    def __init__(self, title, parent=None, fid=-1, pos=wx.DefaultPosition, size=wx.DefaultSize):
        wx.Frame.__init__(self, parent, fid, title, pos, size)

class App(wx.App):

    def OnInit(self):
        self.f =  MyFrame1(title="你好")
        self.f.Show()
        return True

if __name__=="__main__":
    app = App()
    app.MainLoop()
