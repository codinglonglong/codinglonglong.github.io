def checkconnection():
    proxies = {"http": "http://192.168.1.254:8228", "https": "http://192.168.1.254:8228"}
    try:
        r1 = requests.get("http://www.facebook.com",
                          proxies=proxies,
                          timeout=5)
        if str(r1.status_code) == "200":
            return True
        return False
    except Exception as e:
        return False

注意

如果不需要测试代理，去掉 proxies=proxies， 就可以了。
